Variables

Strings

let name = 'Alan Turing';
name[0];                       // 'A'
name[1];                       // 'l'
name[name.length - 1];         // 'g'
'5' + '5'                      // '55'

A leading plus converts a string into a number

const userInput = '19';        // '19'
const age = +userInput;        //  19
const nextYear = age + 1;      //  20

Template strings can include variables concisely

const m = `Age next year: ${nextYear}`;

Functions

function
declaration

  document.body.innerText = text;

arrow
function

const showResult = (text) => {
  document.body.innerText = text;
}

  return 2 * Math.PI * r;
}

Arrow functions make concise callbacks.

Objects

const detective = {
  firstName: 'Sherlock',
  lastName: 'Holmes',
};
const brother = {
  ...detective,
  firstName: 'Mycroft',
};
detective.firstName; // 'Sherlock'
brother.lastName;    // 'Holmes'

Data Processing

const myList = [4, 0, 1, 3];
const sum = (a, b) => a + b;
 
myList.filter(x => x > 2)
→
[4, 3]
myList.map(x => x * 2)
→
[8, 0, 2, 6]
myList.reduce(sum)
→
8
myList.forEach((value) => {
  console.log(value);
});
myList.every(x => x > 5)
→
false
myList.some(x => x > 2)
→
true
myList.join(':')
→
'4:0:1:3'
myList.fill(0)
→
(fills list with 0s)

Copying Arrays

const list1 = [1, 2, 3];
const list2 = [4, 5, 6];

const list3 = [...list1];  // [1, 2, 3];
list1 === list3;           // false

// [1, 2, 3, 4, 5, 6]
const list4 = [...list1, ...list2];
